основи

ObjectMapper і конфігурація

ObjectMapper — центральний клас Jackson. Thread-safe після конфігурації. Один instance на застосунок.

// Spring Boot автоматично конфігурує ObjectMapper // Налаштування через application.properties: spring.jackson.serialization.write-dates-as-timestamps=false spring.jackson.deserialization.fail-on-unknown-properties=false spring.jackson.default-property-inclusion=non_null // Або через @Bean (override autoconfig) @Bean public ObjectMapper objectMapper() { return new ObjectMapper() .registerModule(new JavaTimeModule()) .disable(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS) .disable(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES) .setSerializationInclusion(JsonInclude.Include.NON_NULL) .enable(MapperFeature.DEFAULT_VIEW_INCLUSION); } // Часте використання напряму String json = mapper.writeValueAsString(obj); MyDto dto = mapper.readValue(json, MyDto.class); List<MyDto> list = mapper.readValue(json, new TypeReference<List<MyDto>>() {});
ObjectMapper потокобезпечний для читання/запису, але не для конфігурації. Конфігуруй один раз при створенні, потім тільки використовуй.

Корисні Jackson анотації

// Ім'я поля в JSON @JsonProperty("user_name") private String userName; // Ігнорувати поле @JsonIgnore private String password; // Ігнорувати на рівні класу @JsonIgnoreProperties({"internalId", "createdBy"}) class UserDto { ... } // Ігнорувати невідомі поля (при десеріалізації) @JsonIgnoreProperties(ignoreUnknown = true) // Включати тільки не-null поля @JsonInclude(JsonInclude.Include.NON_NULL) // Аліас для десеріалізації @JsonAlias({"userName", "user_name", "login"}) // Creator для immutable класів (record-like) @JsonCreator public UserDto(@JsonProperty("id") long id, @JsonProperty("name") String name) { ... } // Raw value (String що вже є JSON) @JsonRawValue private String metadata; // {"key":"val"} → вставляється як-є
дати

Дати і JavaTimeModule

Jackson за замовчуванням не вміє серіалізувати LocalDate, LocalDateTime, ZonedDateTime. Потрібен JavaTimeModule.

// Залежність // implementation 'com.fasterxml.jackson.datatype:jackson-datatype-jsr310' // (входить в spring-boot-starter-web автоматично) // Реєстрація модуля mapper.registerModule(new JavaTimeModule()); mapper.disable(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS); // без disable → LocalDate серіалізується як [2024,1,15] (масив!) // Формат в application.properties (Spring Boot) spring.jackson.serialization.write-dates-as-timestamps=false spring.jackson.date-format=yyyy-MM-dd'T'HH:mm:ss // Кастомний формат для конкретного поля @JsonFormat(shape = JsonFormat.Shape.STRING, pattern = "dd.MM.yyyy") private LocalDate birthDate; // record — Jackson 2.12+ підтримує нативно record UserDto(long id, String name, LocalDate birthDate) {}
custom

Custom serializer/deserializer

// Custom serializer class MoneySerializer extends StdSerializer<Money> { MoneySerializer() { super(Money.class); } @Override public void serialize(Money m, JsonGenerator gen, SerializerProvider p) throws IOException { gen.writeStartObject(); gen.writeNumberField("amount", m.getAmount()); gen.writeStringField("currency", m.getCurrency().getCode()); gen.writeEndObject(); } } // Custom deserializer class MoneyDeserializer extends StdDeserializer<Money> { @Override public Money deserialize(JsonParser p, DeserializationContext ctx) throws IOException { ObjectNode node = p.getCodec().readTree(p); return new Money(node.get("amount").decimalValue(), node.get("currency").asText()); } } // Реєстрація SimpleModule module = new SimpleModule(); module.addSerializer(Money.class, new MoneySerializer()); module.addDeserializer(Money.class, new MoneyDeserializer()); mapper.registerModule(module); // Або на полі: @JsonSerialize(using = MoneySerializer.class) @JsonDeserialize(using = MoneyDeserializer.class) private Money price;
@JsonView

@JsonView — різні представлення одного класу

Дозволяє включати різні поля залежно від контексту. Замість окремих DTO для кожного endpoint.

// 1. Визначаємо Views (маркерні інтерфейси) class Views { interface Public {} interface Internal extends Public {} // успадковує Public поля } // 2. Позначаємо поля class UserDto { @JsonView(Views.Public.class) public long id; @JsonView(Views.Public.class) public String name; @JsonView(Views.Internal.class) public String email; // тільки для Internal view public String password; // взагалі не серіалізується } // 3. У контролері @GetMapping("/public/user/{id}") @JsonView(Views.Public.class) public UserDto getPublic(@PathVariable long id) { ... } // → {"id":1, "name":"Alice"} @GetMapping("/internal/user/{id}") @JsonView(Views.Internal.class) public UserDto getInternal(@PathVariable long id) { ... } // → {"id":1, "name":"Alice", "email":"alice@example.com"}
поліморфізм

Поліморфізм і @JsonTypeInfo

Коли поле може бути одним з кількох підтипів — Jackson потрібна підказка про тип для десеріалізації.

// Base class @JsonTypeInfo(use = JsonTypeInfo.Id.NAME, property = "type") @JsonSubTypes({ @JsonSubTypes.Type(value = EmailNotification.class, name = "email"), @JsonSubTypes.Type(value = SmsNotification.class, name = "sms") }) abstract class Notification { ... } // JSON: // {"type":"email", "to":"alice@example.com", "subject":"..."} // → десеріалізується в EmailNotification // Альтернатива: @JsonTypeInfo(use = CLASS) // {"@class":"com.example.EmailNotification", ...} // Небезпечно для зовнішніх API (deserialization gadgets)
@JsonTypeInfo(use = CLASS) — небезпечно в публічних API. Зловмисник може передати будь-який клас з classpath. Використовуй use = NAME з явним whitelist через @JsonSubTypes.
антипатерни

Типові помилки

1. Новий ObjectMapper на кожен виклик

// ПОГАНО: ObjectMapper дорогий у створенні void process(String json) throws Exception { new ObjectMapper().readValue(json, Dto.class); // кожен раз! } // ДОБРЕ: singleton через @Autowired @Autowired ObjectMapper mapper;

2. FAIL_ON_UNKNOWN_PROPERTIES = true (default)

// Якщо API повернув нове поле → UnrecognizedPropertyException // Завжди вимикай для клієнтів зовнішніх API mapper.disable(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES); // або @JsonIgnoreProperties(ignoreUnknown = true)

3. TypeReference без {} — загублений type parameter

// ПОГАНО: type erasure, втрачається List<Dto> mapper.readValue(json, new TypeReference<List<Dto>>(0)); // не анонімний клас! // ДОБРЕ: анонімний підклас зберігає generic type mapper.readValue(json, new TypeReference<List<Dto>>() {});
На співбесіді: ObjectMapper thread-safe і singleton, WRITE_DATES_AS_TIMESTAMPS=false для ISO dates, @JsonView для різних представлень, TypeReference для generic types, @JsonTypeInfo з NAME (не CLASS) для безпеки.
